
					function initCalendar()
					{
						if (typeof(SetCache) != "undefined" && typeof(SetCalendar) != "undefined")
						{
							var	cacheKey	=	'_UTK0101_Calendar';
							var	cacheData	=	GetCache(cacheKey);
							if (cacheData != null)
							{
								_cevent	=	JSON.parse(cacheData);
								SetCalendar(0);
							}
							else
							{
								DoPost("", _vr + "UTK0101_/GET_CALENDAR_EVENTS", function (data, text) {SetCache(cacheKey, JSON.stringify(_cevent), 10);SetCalendar(0);});
							}
						}
						else
							window.setTimeout(function(){initCalendar();}, 100);
					}
					initCalendar();
